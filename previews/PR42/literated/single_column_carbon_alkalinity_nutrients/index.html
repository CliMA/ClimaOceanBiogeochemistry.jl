"""

Arguments:
- `grid::RectilinearGrid`: The model grid.
- `schmidt_number_dic::Field{Center, Center, Nothing}`: The computed Schmidt number for DIC.
- `temperature::Field{Center, Center, Nothing}`: Temperature in degrees Celsius.
- `kЛўб¶ң::CarbonCoefficientParameters`: The parameters for the Schmidt number calculation.

@kernel function compute_schmidt_dic!(
    grid,
    schmidt_number_dic,
    temperature,
    kЛўб¶ң = CarbonCoefficientParameters(
            aвӮҖ = 2116.8,
            aвӮҒ = 136.25,
            aвӮӮ = 4.7353,
            aвӮғ = 9.2307e-2,
            aвӮ„ = 7.555e-4,

         kЛўб¶ң.aвӮҖ -
         kЛўб¶ң.aвӮҒ * temperature[i, j, k] +
         kЛўб¶ң.aвӮӮ * temperature[i, j, k]^2 -

"""
    compute_piston_velocity(
        grid,
        piston_velocity,
        surface_wind_speed,
        exchange_coefficient,
        schmidt_number,
        )

Compute the piston velocity for gas exchange at the ocean surface.

Arguments:
- `grid::RectilinearGrid`: The model grid.
- `piston_velocity::Field{Center, Center, Nothing}`: The computed piston velocity.
- `surface_wind_speed::Field{Center, Center, Nothing}`: The wind speed at the ocean surface.
- `exchange_coefficient::Float64`: The gas exchange coefficient.
- `schmidt_number::Field{Center, Center, Nothing}`: The Schmidt number.
"""
@kernel function compute_piston_velocity!(
    grid,
    piston_velocity,
    surface_wind_speed,
    exchange_coefficient,
    schmidt_number
    )

    i, j = @index(Global, NTuple)

    piston_velocity[i, j, 1] = exchange_coefficient[i, j, 1] *
                               surface_wind_speed[i, j, 1]^2 /
                               sqrt(schmidt_number[i, j, 1])
end

- `grid::RectilinearGrid`: The model grid.
- `reference_density::Float64`: The reference density of seawater.
- `COвӮӮ_flux::Field{Center, Center, Nothing}`: The computed COвӮӮ flux.

- `oceanic_pCOвӮӮ::Field{Center, Center, Nothing}`: The partial pressure of COвӮӮ in the ocean.
- `atmospheric_COвӮӮ_solubility::Field{Center, Center, Nothing}`: The solubility of COвӮӮ in the atmosphere.
- `oceanic_COвӮӮ_solubility::Field{Center, Center, Nothing}`: The solubility of COвӮӮ in the ocean.

Notes:
The convention is that a positive flux is upwards (outgassing), and a negative flux is downwards (uptake).
"""
@kernel function compute_COвӮӮ_flux!(
    grid,
    reference_density,
    COвӮӮ_flux,
    piston_velocity,
    atmospheric_pCOвӮӮ,
    oceanic_pCOвӮӮ,
    atmospheric_COвӮӮ_solubility,
    oceanic_COвӮӮ_solubility,
    )
    i, j = @index(Global, NTuple)

    # compute COвӮӮ flux (-ve for uptake, +ve for outgassing since convention is +ve upwards)
    COвӮӮ_flux[i, j, 1] = - piston_velocity[i, j, 1] * (
                 atmospheric_pCOвӮӮ[i, j, 1] * atmospheric_COвӮӮ_solubility[i, j, 1] -
                 oceanic_pCOвӮӮ[i, j, 1]     * oceanic_COвӮӮ_solubility[i, j, 1]
            ) * reference_density # Convert mol kgвҒ»В№ m sвҒ»В№ to mol mвҒ»ВІ sвҒ»В№
end
